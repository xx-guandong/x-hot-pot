import Note from "../components/mdx/Note";
import MyLayoutComponent from "../components/mdx/DefaultLayout";
import QuickNav from "../components/mdx/QuickNav";
import Page from "../lib/prisma/Page";

export default ({ children, data }) => (
  <MyLayoutComponent data={data}>{children}</MyLayoutComponent>
);


export async function getServerSideProps() {
   // Fetch data from external API
  Page.increasePageById({ id: 2 });
  const result = await Page.findPageById({ id: 2 });
  // Pass data to the page via props
  return { props: { data: result } };
}


## 1. Install

Install Nestjs with [typescript](https://www.typescriptlang.org/docs/handbook/typescript-from-scratch.html), you can choose one of these methods.

```
npx create-next-app@latest --typescript
# or
yarn create next-app --typescript
# or
pnpm create next-app -- --typescript
```

- [npx introduction](https://github.com/npm/cli)
- [yarn introduction](https://yarnpkg.com/getting-started)
- [pnpm introduction](https://pnpm.io/installation)
